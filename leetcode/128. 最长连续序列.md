# 128. 最长连续序列


#### 首次提交

```
class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        unordered_set<int> s(nums.begin(), nums.end());
        int maxLen = 0;
        for (int num : nums) {
            auto it = s.find(num - 1);
            if (it != s.end()) {
                continue;
            }
            int len = 1;
            int tempNum = num;
            while (s.find(++tempNum) != s.end()) {
                len++;
            }
            if (len > maxLen) {
                maxLen = len;
            }
        }
        return maxLen;
    }
};
```
