# 199. 二叉树的右视图

### 初次提交
```
class Solution {
public:
    vector<int> rightSideView(TreeNode* root) {
        // 广度优先遍历
        vector<int> v;
        if (!root) {
            return v;
        }
        queue<TreeNode*> q;
        q.push(root);
        int currentNums = 1;
        int nextNums = 0;
        while (!q.empty()) {
            TreeNode* node = q.front();
            q.pop();
            currentNums --;
            if (node->left) {
                q.push(node->left);
                nextNums++;
            }
            if (node->right) {
                q.push(node->right);
                nextNums++;
            }
            if (currentNums == 0) {
                v.push_back(node->val);
                currentNums = nextNums;
                nextNums = 0;
            }
        }
        return v;
    }
};
```
