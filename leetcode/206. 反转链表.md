# 206. 反转链表

#### 初次提交

```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        std::stack<ListNode*> stk;
        while (head != nullptr) {
            stk.push(head);
            head = head->next;
        }
        ListNode* res = nullptr;
        ListNode* rare = nullptr;
        while (!stk.empty()) {
            ListNode* node = stk.top();
            node->next = nullptr;
            if (res == nullptr) {
                res = node;
                rare = res;
            }
            else {
                rare->next = node;
                rare = node;
            }
            stk.pop();
        }
        return res;
    }
};
```

#### 优化

能不能优化一下空间复杂度
```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if (head == nullptr) return nullptr;
        ListNode* newHead = head;
        head = head->next;
        newHead->next = nullptr;
        ListNode* headTemp;
        while (head != nullptr) {
            headTemp = head->next;
            head->next = newHead;
            newHead = head;
            head = headTemp;
        }
        return newHead;
    }
};
```
#### 分析
时间复杂度：O(n)，空间复杂度O(1)，但是代码看起来可读性不高，再改改


#### 第三次提交
```
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        // 算法思想：遍历链表过程中，将当前节点指向前一个节点
        ListNode* pre = nullptr; // 用于记录前一个节点的指针
        ListNode* temp = nullptr; // 用于记录遍历的下一个节点的指针
        while (head) {
            temp = head->next; // 用于记录遍历的下一个节点
            head->next = pre;
            pre = head;
            head = temp;
        }
        // 遍历结束时，pre为新链表表头
        return pre;
    }
};
```

#### 分析：
还算不错，比官方给出的题解的内存消耗要小，少了一个指针的定义，以及不在while中做声明
