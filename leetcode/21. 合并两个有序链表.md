# 21. 合并两个有序链表


### 初次提交
```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        // 双指针 O(n) O(1)
        // 1. 将其中一个合并到另一个
        // 2. 将2个链表合并到新的链表
        if (!list1 || !list2) {
            return list1 == nullptr ? list2 : list1;
        }
        ListNode *temp;
        ListNode* res = list2;
        while (list1 != nullptr && list2 != nullptr) {
            if (list2->val <= list1->val) {
                if (list2->next && list2->next->val > list1->val) {
                    temp = list1->next;
                    list1->next = list2->next;
                    list2->next = list1;
                    list1 = temp;
                } else if (list2->next == nullptr) {
                    list2->next = list1;
                    break;
                }
                list2 = list2->next;
            } else {
                temp = list1->next;
                list1->next = list2;
                list2 = list1;
                list1 = temp;
                res = list2;
            }
            if (list2->next == nullptr) {
                list2->next = list1;
                break;
            } 
        }
        return res;
    }
};
```

### 结果


### 分析
时间复杂度O(n)，空间复杂度O(1)，但是代码不容易理解，是将list1合并至list2中

### 优化

考虑将list1 与 list2的较小的节点合并至新的节点list3中，这样更容易理解

### 第二次提交

```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        // 双指针 O(n) O(1)
        // 2. 将2个链表合并到新的链表
        ListNode* head = nullptr;
        ListNode* rear = nullptr;
        ListNode* min = nullptr;
        while (list1 != nullptr && list2 != nullptr) {
            // 合并直到一个链表为空
            bool minCompare = list1->val < list2->val;
            min = minCompare ? list1 : list2;
            if (!rear) {
                rear = min;
                head = min;
            } else {
                rear->next = min;
                rear = rear->next;
            }
            if (minCompare) {
                list1 = list1->next;
            } else {
                list2 = list2->next;
            }
            rear->next = nullptr;
        }
        // 合并剩余
        if (rear != nullptr) {
            rear->next = list1 == nullptr ? list2 : list1;
        } else {
            head = list1 == nullptr ? list2 : list1;
        }

        return head;
    }
};
```
