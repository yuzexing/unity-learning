# 560. 和为 K 的子数组

### 初次提交
```
class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        // 一眼先暴力搜索
        int num = 0;
        for (int i = 0; i < nums.size(); ++i) {
            int total = 0;
            for (int j = i; j < nums.size(); ++j) {
                total += nums[j];
                if (k == total) {
                    num++;
                }
            }
        }
        return num;
    }
};
```
