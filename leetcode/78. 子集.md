# 78. 子集


### 初次提交
```
class Solution {
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        vector<vector<int>> res;
        vector<int> temp;
        int size = nums.size();
        // int length = pow(2, size);
        for (int i = 0; i < (1 << size); i++) {
            int mask = i;
            temp.clear();
            for (int index = 0; index < nums.size(); index++) {
                if (i & (1 << index)) {
                    temp.push_back(nums[index]);
                }
            }
            res.push_back(temp);
        }

        return res;
    }
};
```
